<script>
    function formatNumber(num) {
        return Math.floor(num).toLocaleString();
    }

    function calculate() {
        const dailyInvest = parseFloat(document.getElementById('dailyInvest').value);
        const rate = parseFloat(document.getElementById('rate').value);
        const swapPer10k = parseFloat(document.getElementById('swapPer10k').value);
        const swapDecay = parseFloat(document.getElementById('swapDecay').value) / 100;
        const leverage = parseFloat(document.getElementById('leverage').value);
        const years = parseInt(document.getElementById('years').value);

        let balance = 0;
        let totalPrincipal = 0;
        let cumulativeSwap = 0;

        const tbody = document.getElementById('results');
        tbody.innerHTML = '';

        for (let y = 1; y <= years; y++) {
            const currentSwapPer10k = swapPer10k * Math.pow(1 - swapDecay, y - 1);
            const currentSwapPerUnit = currentSwapPer10k / 10000;
            
            let yearlySwap = 0;

            for (let d = 0; d < 365; d++) {
                const dailySwap = balance * currentSwapPerUnit;
                yearlySwap += dailySwap;
                cumulativeSwap += dailySwap;

                totalPrincipal += dailyInvest;

                // 投資元本とスワップにレバレッジをかけて買付力を計算
                const buyingPower = (dailyInvest + dailySwap) * leverage;
                const boughtCurrency = buyingPower / rate;
                
                balance += boughtCurrency;
            }

            const lotCount = (balance / 10000).toFixed(2);
            const eoyDailySwap = balance * currentSwapPerUnit;
            const monthlySwapAvg = yearlySwap / 12;
            const assetValue = totalPrincipal + cumulativeSwap;

            const tr = document.createElement('tr');
            tr.innerHTML = `
                <td style="text-align:center;">${y}年後</td>
                <td>${formatNumber(balance)}</td>
                <td>${lotCount}</td>
                <td>${formatNumber(totalPrincipal)}</td>
                <td>${formatNumber(eoyDailySwap)}</td>
                <td>${formatNumber(monthlySwapAvg)}</td>
                <td>${formatNumber(yearlySwap)}</td>
                <td style="font-weight:bold; color:#27ae60;">${formatNumber(assetValue)}</td>
            `;
            tbody.appendChild(tr);
        }
    }

    // 初期表示時に1回計算を実行する
    window.onload = calculate;
</script>
