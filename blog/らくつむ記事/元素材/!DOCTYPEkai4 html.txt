<script>
    function formatNumber(num) {
        return Math.floor(num).toLocaleString();
    }

    function calculate() {
        const reinvestMode = document.getElementById('reinvestMode').value;
        const initialCapital = parseFloat(document.getElementById('initialCapital').value) || 0;
        const initialBalance = parseFloat(document.getElementById('initialBalance').value) || 0;
        const dailyInvest = parseFloat(document.getElementById('dailyInvest').value) || 0;
        const rate = parseFloat(document.getElementById('rate').value) || 1;
        const swapPer10k = parseFloat(document.getElementById('swapPer10k').value) || 0;
        const swapDecayAnn = parseFloat(document.getElementById('swapDecay').value) / 100 || 0;
        const leverage = Math.max(0.1, parseFloat(document.getElementById('leverage').value) || 1);
        const months = parseInt(document.getElementById('months').value) || 12;

        const swapDecayMonth = swapDecayAnn / 12;

        let balance = initialBalance;
        let totalPrincipal = 0;
        let cumulativeSwap = 0;

        const tbody = document.getElementById('results');
        tbody.innerHTML = '';

        for (let m = 1; m <= months; m++) {
            const currentSwapPer10k = swapPer10k * Math.pow(1 - swapDecayMonth, m - 1);
            const currentSwapPerUnit = currentSwapPer10k / 10000;
            
            let monthlySwapTotal = 0;

            for (let d = 0; d < 30; d++) {
                const dailySwap = balance * currentSwapPerUnit;
                monthlySwapTotal += dailySwap;
                cumulativeSwap += dailySwap;
                totalPrincipal += dailyInvest;

                let buyingPower = 0;
                if (reinvestMode === 'compound') {
                    buyingPower = (dailyInvest + dailySwap) * leverage;
                } else {
                    buyingPower = dailyInvest * leverage;
                }
                
                const boughtCurrency = buyingPower / rate;
                balance += boughtCurrency;
            }

            const lotCount = (balance / 10000).toFixed(2);
            const eomDailySwap = balance * currentSwapPerUnit;
            const yearlySwapPace = eomDailySwap * 365;
            
            // 口座清算価値 (有効証拠金)
            const totalAsset = initialCapital + totalPrincipal + cumulativeSwap;
            
            // 必要証拠金
            const requiredMargin = (balance * rate) / leverage;
            
            // 証拠金維持率
            let maintenanceRateStr = '-';
            if (requiredMargin > 0) {
                const maintenanceRate = (totalAsset / requiredMargin) * 100;
                maintenanceRateStr = maintenanceRate.toFixed(1) + '%';
            }

            const tr = document.createElement('tr');
            tr.innerHTML = `
                <td class="month-col">${m}ヶ月目</td>
                <td class="asset-val">${formatNumber(totalAsset)}</td>
                <td class="maintenance-val">${maintenanceRateStr}</td>
                <td>${formatNumber(balance)}</td>
                <td>${lotCount}</td>
                <td>${formatNumber(totalPrincipal)}</td>
                <td class="highlight-col">${formatNumber(eomDailySwap)}</td>
                <td>${formatNumber(monthlySwapTotal)}</td>
                <td>${formatNumber(yearlySwapPace)}</td>
            `;
            tbody.appendChild(tr);
        }
    }

    window.onload = calculate;
</script>
